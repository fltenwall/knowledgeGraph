# TreeSelector

树型选择器。

## 何时使用

<ul>
  <li>左右层次的树结构列表，可以可以完整展现其中的层级关系。</li>
  <li>基于 antd 的 Tree 进行封装。</li>
</ul>

## 基本使用

<Playground>
  {
    () => {
      const data = [
        { 
          title: '0-0', 
          key: '0-0', 
          children: [
            { title: '0-0-0', key: '0-0-0' },
            { title: '0-0-1', key: '0-0-1' },
            { title: '0-0-2', key: '0-0-2' }
          ]
        },
        { 
          title: '0-1', 
          key: '0-1',
          children: [
            { title: '0-1-0', key: '0-1-0' },
            { title: '0-1-1', key: '0-1-1' },
            { title: '0-1-2', key: '0-1-2' }
          ]
        },
        { 
          title: '0-2', 
          key: '0-2',
          children: [
            { title: '0-2-0', key: '0-2-0' },
            { title: '0-2-1', key: '0-2-1' },
            { title: '0-2-2', key: '0-2-2' }
          ]
        }
      ];

      return (
        <TreeSelector defaultExpandedKey="0-0" data={data} span={12} />
      )
    }
  }
</Playground>

## 受控操作示例

<Playground>
  {
    () => {
      const data = [
        { 
          title: '0-0', 
          key: '0-0', 
          children: [
            { title: '0-0-0', key: '0-0-0' },
            { title: '0-0-1', key: '0-0-1' },
            { title: '0-0-2', key: '0-0-2' }
          ]
        },
        { 
          title: '0-1', 
          key: '0-1',
          children: [
            { title: '0-1-0', key: '0-1-0' },
            { title: '0-1-1', key: '0-1-1' },
            { title: '0-1-2', key: '0-1-2' }
          ]
        },
        { 
          title: '0-2', 
          key: '0-2',
          children: [
            { title: '0-2-0', key: '0-2-0' },
            { title: '0-2-1', key: '0-2-1' },
            { title: '0-2-2', key: '0-2-2' }
          ]
        }
      ];

      const [checkedList, setValue] = useState(["0-0-0", "0-1-0", "0-2-0"]);

      const onChange = value => {
        setValue(value);
      }

      return (
        <TreeSelector defaultExpandedKey="0-0" data={data} checkedList={checkedList} onChange={onChange} />
      )
    }
  }
</Playground>
